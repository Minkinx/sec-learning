# 业务逻辑漏洞

## 概述

业务逻辑漏洞是利用应用程序业务流程设计缺陷的安全问题，而非传统意义上的技术漏洞。这类漏洞难以通过自动化扫描工具检测，因为它们涉及对业务规则的理解和操作顺序的滥用。典型场景包括支付逻辑、优惠券系统、账户操作和权限控制等。

## OTP（一次性密码）绕过

### 常见OTP实现缺陷

```text
# 响应篡改
POST /api/verify-otp
{"phone": "13800138000", "otp": "123456"}
# 正常响应：{"success": false, "message": "Invalid OTP"}
# 攻击者修改响应：{"success": true} → 绕过验证

# OTP可重用
场景：用户收到OTP，输入错误后重新输入，OTP未失效
攻击：固定OTP尝试所有可能值（4位=10000次组合）

# 无锁定机制
POST /api/verify-otp
{"phone": "13800138000", "otp": "0000"} → 失败
{"phone": "13800138000", "otp": "0001"} → 失败
# 无限制重试，可暴力遍历

# OTP仅4位数字
10000种组合，每秒100次请求 → 100秒内破解
```

### OTP逻辑竞争条件

```http
# 同时发送多个验证请求
# 请求1
POST /api/verify-otp
{"phone": "13800138000", "otp": "1234"}

# 请求2（同时）
POST /api/verify-otp
{"phone": "13800138000", "otp": "5678"}

# 如果服务器在TOCTOU窗口内未正确处理并发验证，
# 两个请求可能都被接受
```

### OTP绕过防御

```python
# 安全的OTP验证逻辑
import time
from flask_limiter import Limiter
from flask import session

OTP_EXPIRE_SECONDS = 300  # 5分钟过期
MAX_ATTEMPTS = 5
LOCKOUT_MINUTES = 30

def verify_otp(phone, otp_input):
    key = f"otp_attempts:{phone}"
    
    # 检查是否被锁定
    if cache.get(f"lockout:{phone}"):
        return False, "Account locked"
    
    # 获取存储的OTP
    stored_otp = cache.get(f"otp:{phone}")
    if not stored_otp:
        return False, "OTP expired"
    
    # 验证OTP
    if stored_otp != otp_input:
        attempts = cache.incr(key)
        if attempts >= MAX_ATTEMPTS:
            cache.setex(f"lockout:{phone}", LOCKOUT_MINUTES * 60, 1)
            cache.delete(key)
            return False, "Account locked"
        return False, "Invalid OTP"
    
    # 验证成功后清除
    cache.delete(f"otp:{phone}")
    cache.delete(key)
    
    # 生成唯一验证Token
    token = secrets.token_hex(32)
    cache.setex(f"verified:{phone}", 600, token)  # 10分钟内有效
    return True, token
```

### 秒杀/抢购竞争

```python
# 安全的原子操作
import redis

def safe_apply_coupon(user_id, coupon_code):
    r = redis.Redis()
    # 使用Redis原子操作
    key = f"coupon:{coupon_code}"
    # SETNX - 仅当不存在时设置
    acquired = r.setnx(f"lock:{coupon_code}", user_id)
    if not acquired:
        return None, "Already claimed"
    
    try:
        coupon = db.query("SELECT * FROM coupons WHERE code = ?", coupon_code)
        if not coupon:
            return None, "Invalid"
        
        # 使用数据库锁（悲观锁）
        db.execute("BEGIN")
        try:
            db.execute("UPDATE coupons SET used = 1, used_by = ? WHERE code = ? AND used = 0", 
                      user_id, coupon_code)
            if db.affected_rows() == 0:
                db.execute("ROLLBACK")
                return None, "Already used"
            db.execute("COMMIT")
        except:
            db.execute("ROLLBACK")
            raise
    finally:
        r.delete(f"lock:{coupon_code}")
```

## 防御措施

### 服务端验证原则

```python
# 所有业务逻辑必须在服务端验证
def process_order(user, cart, coupon_code):
    # 1. 验证用户身份（不可依赖前端传的user_id）
    if not authenticate_user(session):
        raise AuthError()
    
    # 2. 重新计算价格（不可信赖前端价格）
    total = 0
    for item in cart['items']:
        product = get_product_from_db(item['product_id'])
        item['price'] = product['price']  # 覆盖前端传来的价格
        total += product['price'] * item['quantity']
    
    # 3. 验证数量（不允许负数或为零）
    if item['quantity'] <= 0:
        raise ValidationError("Invalid quantity")
    
    # 4. 验证库存
    if product['stock'] < item['quantity']:
        raise StockError("Insufficient stock")
    
    # 5. 应用优惠券验证
    if coupon_code:
        coupon = validate_coupon(coupon_code, user)
        total = apply_discount(coupon, total)
    
    # 6. 验证最低金额
    if total < 0:
        raise ValidationError("Invalid total")
    if total < MINIMUM_ORDER:
        raise ValidationError("Below minimum order")
    
    return create_order(user, items, total)
```

@transaction.atomic
def redeem_coupon(user_id, coupon_code):
    coupon = Coupon.objects.select_for_update().get(code=coupon_code)

# 使用幂等性Key防止重复处理
def process_payment(order_id, idempotency_key):
    # 检查是否已处理
    existing = cache.get(f"idempotent:{idempotency_key}")
